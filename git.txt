# создать новый репозиторий
git init

# клонировать репозиторий
git clone https://github.com/ravlyuk/notes.git

# добавить указаный файл в репозиторий
git add program.py

# добавить всё в репозиторий
git add .

# узнать статус
git status

# закоминтить
git commit -m "Текст вашего сообщения о том, что изменено"

# подключить указанный репозиторий
git remote add origin https://github.com/ravlyuk/notes.git

# выполнить синхронизацию с репозиторией
git push -u origin master

# удалить репозиторий
git remote rm origin

# показать текущий репозиторий
git remote -v
